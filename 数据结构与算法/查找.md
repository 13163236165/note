## 二分查找

```c++
int BinarySearch(vector<int> data,int k){
    if(data.size() == 0)
        return -1;
    
    int sz = data.size();
    int l = 0,r = sz - 1;
    while(l <= r){
        int mid = (l + r) >> 1;
        if(data[mid] > k)
            r = mid - 1;
        else if(data[mid] < k)
            l = mid + 1;
        else
            return mid;
    }
    
    return -1;
}
```

### 1.查找元素第一次出现的下标

```c++
int GetFirstK(vector<int> data,int k){
    if(data.size() == 0)
        return -1;
    
    int sz = data.size();
    int l = 0,r = sz - 1;
    while(l <= r){
        int mid = (l + r) >> 1;
        if(data[mid] > k)
            r = mid - 1;
        else if(data[mid] < k)
            l = mid + 1;
        else{
            if(mid == 0 || data[mid - 1] != k) 
                return mid;
            else 
                r = mid - 1;
        }
    }
    
    return -1;
}
```

### 2.查找元素最后一次出现的下标

```c++
int GetLastK(vector<int> data,int k){
    if(data.size() == 0)
        return -1;
    
    int sz = data.size();
    int l = 0,r = sz - 1;
    while(l <= r){
        int mid = (l + r) >> 1;
        if(data[mid] > k)
            r = mid - 1;
        else if(data[mid] < k)
            l = mid + 1;
        else{
            if(mid == sz - 1 || data[mid + 1] != k)
                return mid;
            else
                l = mid + 1;
        }
    }
    
    return -1;
}
```
